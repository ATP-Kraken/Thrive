// Quit the game //
o GuiObject "Background/MainMenuInteractive/ExitGameButton"{
    s{
        [@Listener="OnClick"]
        bool QuitIt(GuiObject@ instance){
            // Close the game //
            cast<LeviathanApplication>(GetThriveGame()).MarkAsClosing();

        }
        @%};
}
